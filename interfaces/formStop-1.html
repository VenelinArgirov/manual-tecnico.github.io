import { Component, OnDestroy, OnInit } from '@angular/core';
import { RouteService } from '../route.service';
import { AbstractControl, FormArray, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Route } from '../route';
import { ActivatedRoute, convertToParamMap } from '@angular/router';
import { Stop } from '../stop';
import { TransportType } from '../transport-type';
import { StopService } from '../stop.service';
import { DayType } from '../day-type';
import { LinkedStop } from "../linked-stop";
import { LinkedStopService } from '../linked-stop.service';
import { Guid } from 'guid-typescript';
import { map, Observable } from 'rxjs';
import { MapDirectionsService } from '@angular/google-maps';
import { Router } from '@angular/router';

@Component({
  selector: 'app-route-edit',
  templateUrl: './route-edit.component.html',
  styleUrls: ['./route-edit.component.css']
})
/**
 * Class for the component of the page of Route Editing.
 */
export class RouteEditComponent implements OnInit, OnDestroy {
  /**
   * Options to be used by the Google-Map field.
   */
  mapOptions: google.maps.MapOptions = {
    mapTypeId: 'roadmap',
    disableDefaultUI: true,
    zoomControl: true,
    clickableIcons: false,
    disableDoubleClickZoom: true,
    draggable: false,
  };
  /**
   * Options to be used by the Directions-Renderer field.
   */
  directionsOptions: google.maps.DirectionsRendererOptions = {
    suppressMarkers: true,
    suppressInfoWindows: true,
  };
  /**
   * Options to be used by the Map-Marker field.
   */
  markerOptions: google.maps.MarkerOptions = { draggable: false };

  /**
   * Zoom level for the map to use at the start.
   */
  zoom = 11;
  /**
   * Coordinates for the map to be centered on.
   */
  center!: google.maps.LatLngLiteral;
  /**
   * List of markers that represents the stops of the route to edit.
   */
  markers: Array<google.maps.LatLng> = new Array<google.maps.LatLng>();
  /**
   * Holds the result from the call of the Directions API made from the list of markers.
   */
  directionsResults$!: Observable<google.maps.DirectionsResult | undefined>;

  /**
   * TransportType for reverse mapping in the html.
   */
  TransportType = TransportType;
  /**
   * DayType for reverse mapping in the html.
   */
  DayType = DayType;

  /**
   * Id of the route.
   */
  id: string = '';
  /**
   * Route to edit.
   */
  route!: Route;
  /**
   * All the existent stops.
   */
  AllStops: Stop[] = [];
  /**
   * All the values of the TransportType enum.
   */
  transports: number[] = Object.keys(TransportType).map(v => Number(v)).filter(v => Number.isFinite(v));
  /**
   * All the values of the DayType enum.
   */
  days: number[] = Object.keys(DayType).map(v => Number(v)).filter(v => Number.isFinite(v));
  /**
   * @ignore
   */
  private sub: any;

  /**
   * Form data to be viewed in the page.
   */
  editRouteForm = this.formBuilder.group({
    transportNumber: this.formBuilder.control<number | null>(null),
    name: ['', Validators.required],
    companyName: ['', Validators.required],
    transportType: this.formBuilder.control<number | null>(null, Validators.required),
    description: '',
    stops: this.formBuilder.array([]),
    day: this.formBuilder.control<number | null>(null, Validators.required)
  });

  /**
   * Constructor for the component.
   * Gets the route whose Id matches the one in the URL and passes its info to the form.
   * Also gets all the existent stops.
   * @param {ActivatedRoute} actRoute Gives information about the current route.
   * @param {RouteService} routeService Service that communicates with the server and has various operations about Routes.
   * @param {StopService} stopService Service that communicates with the server and has various operations about Stops.
   * @param {LinkedStopService} linkedStopService Service that communicates with the server and has various operations about LinkedStops.
   * @param {MapDirectionsService} mapDirectionsService Google service of the Directions API.
   * @param {FormBuilder} formBuilder Builds form data.
   * @param {Router} router Router.
   */
  constructor(private routeService: RouteService, private stopService: StopService,
    private linkedStopService: LinkedStopService, private formBuilder: FormBuilder,
    private actRoute: ActivatedRoute, private mapDirectionsService: MapDirectionsService,
    private router: Router) { }

  ngOnInit(): void {
    this.sub = this.actRoute.params.subscribe(async params => {
      this.id = String(params['id']);
      let p1 = new Promise<void>((resolve, reject) => {
        this.routeService.getRoute(this.id).subscribe(result => {
          var stops = new Array<LinkedStop>;
          result.stops?.forEach(stop => {
            stops.push(new LinkedStop(stop.id, stop.stopId, stop.routeId, stop.time, stop.day, stop.stop, stop.route));
          });
          this.route = new Route(result.routeId, result.transportNumber, result.name, result.description, result.transportType, result.companyName, stops);
          if (this.route) {
            this.editRouteForm.controls.transportNumber.patchValue(this.route.transportNumber);
            this.editRouteForm.controls.name.patchValue(this.route.name);
            this.editRouteForm.controls.companyName.patchValue(this.route.companyName);
            this.editRouteForm.controls.transportType.patchValue(this.route.transportType);
            this.editRouteForm.controls.description.patchValue(this.route.description);
            if (this.route.stops) {
              this.editRouteForm.controls.stops.patchValue(this.route.stops);
              this.editRouteForm.controls.day.patchValue(this.route.stops[0].day);
            }
            stops.forEach(stop => {
              const stopForm = this.formBuilder.group({
                stopId: [stop.stop?.stopId, Validators.required],
                stopTime: [stop.getHoursAndMinutesString(), Validators.required]
              });
              this.stops.push(stopForm);
            });
            resolve();
          }
        }, error => console.error(error));
      });
      let p2 = new Promise<void>((resolve, reject) => {
        this.stopService.getStops().subscribe(result => {
          this.AllStops = result;
          resolve();
        }, error => console.error(error));
      });
      await Promise.all([p1, p2])
      this.updateMap();
    });
  }

  ngOnDestroy() {
    this.sub.unsubscribe();
  }

  /**
   * Adds a form to the page that lets the user select the details for a stop in the route.
   */
  addStop() {
    const stopForm = this.formBuilder.group({
      stopId: ['', Validators.required],
      stopTime: ['', Validators.required]
    });

    this.stops.push(stopForm);
  }

  /**
   * Removes the form of one of the route stops.
   * @param {number} stopIndex The form index to delete.
   */
  removeStop(stopIndex: number) {
    this.stops.removeAt(stopIndex);
    this.updateMap();
  }

  /**
   * Gets all the stop forms in the page.
   */
  get stops() {
    return this.editRouteForm.controls["stops"] as FormArray;
  }

  /**
   * Casts a AbstractControl form to a FormGroup.
   */
  toFormGroup = (form: AbstractControl) => form as FormGroup;

  /**
   * Updates the map in the page by adding/removing markers and linking them together in one route.
   */
  updateMap() {
    this.addMarkers();
    this.linkMarkers();
  }

  /**
   * Iterates through the list of stop forms and creates a new marker from the corresponding stop and its coordinates.
   */
  private addMarkers() {
    this.markers.splice(0);

    this.editRouteForm.value.stops?.forEach(s => {
      var stop = (s as formStop);
      
      this.AllStops.forEach(compStop => {
        if (compStop.stopId == stop.stopId) this.markers.push(new google.maps.LatLng(compStop.location.latitude, compStop.location.longitude));
      });
    });
    this.center = {
      lat: this.markers[Math.floor(this.markers.length / 2)].lat(),
      lng: this.markers[Math.floor(this.markers.length / 2)].lng()
    }
  }

  /**
   * Using the Google Directions API, shows the path the route takes using the first and last marker in the route.
   */
  private linkMarkers() {
    const request: google.maps.DirectionsRequest = {
      origin: this.markers[0],
      destination: this.markers[this.markers.length-1],
      travelMode: google.maps.TravelMode.TRANSIT,
    };
    this.directionsResults$ = this.mapDirectionsService.route(request).pipe(map(response => response.result));
  }

  /**
   * Creates a route object and sends it to the database using the routeService.
   */
  onSubmit(): void {
    let route: Route = new Route(this.route.routeId, this.editRouteForm.controls.transportNumber.value!,
      this.editRouteForm.controls.name.value!, this.editRouteForm.controls.description.value!,
      Number(this.editRouteForm.controls.transportType.value!), this.editRouteForm.controls.companyName.value!
    );

    //existe problema em que nos detalhes pode aparecer mais ou menos paragens, mas na bd est√° sempre o numero correto
    this.routeService.editRoute(route).subscribe();
    this.route.stops?.forEach(linkedStop => this.linkedStopService.deleteLinkedStop(linkedStop.id).subscribe());

    this.editRouteForm.value.stops?.forEach(s => {
      var stop = (s as formStop);
      var date = new Date();
      date.setHours(Number(stop.stopTime.substring(0, 2)));
      date.setMinutes(Number(stop.stopTime.substring(3, 5)));
      let linkedStop: LinkedStop = new LinkedStop(Guid.create().toString(), stop.stopId, route.routeId, date, Number(this.editRouteForm.controls.day.value!));
      this.linkedStopService.addLinkedStop(linkedStop).subscribe();
    });
    this.router.navigate(['/', 'route-details', this.route.routeId]);
  }

  /**
   * Cancels the creation of a new stop. Redirects back to the previous page.
   */
  cancel(): void {
    window.history.back();
  }
}

/**
 * Interface for the stop form
 */
interface formStop {
  stopId: string,
  stopTime: string
}
