import { LinkedStop } from "./linked-stop";

/**
 * Class that represents a route.
 */
export class Route {
  /**
   * Id of the route.
   */
  routeId: string;
  /**
   * Number of the transport that runs the route.
   */
  transportNumber: number | null;
  /**
   * Name of the route
   */
  name: string;
  /**
   * Company that owns the transport of the route
   */
  companyName: string;
  /**
   * The type of the transport of the route
   */
  transportType: number;
  /**
   * Description of the route
   */
  description: string;
  /**
   * The time it takes for a transport to run the route.
   */
  routeDuration!: number;

  /**
   * Constructor for the class.
   */
  constructor(routeId: string, transportNumber: number | null, name: string,
    description: string, transportType: number, companyName: string, public stops?: Array<LinkedStop>) {
    this.routeId = routeId;
    this.transportNumber = transportNumber;
    this.name = name;
    this.companyName = companyName;
    this.description = description;
    this.transportType = transportType;
    if (stops) this.stops = stops;
    this.routeDuration = this.calculateRouteDuration();
  }

  /**
   * Function used to calculate the duration time for a route.
   */
  calculateRouteDuration(): number {
    // tem de se criar uma nova data porque por alguma razao ele queixa-se
    // que getTime() não é uma função se não o fizermos
    if (this.stops) {
      var finishTime = new Date(this.stops[this.stops.length - 1].time).getTime();
      var startTime = new Date(this.stops[0].time).getTime();
      return Math.floor((finishTime - startTime) / 60000);
    }
    return 0;
  }
}
