Future<RouteTripInfoMessage> getRouteTripInfoFromFeedMessage(String feedMessageId) async {
  /// Defining the URL for the routes API
  final url = Uri.parse(
      'https://traveller-server.azurewebsites.net/api/FeedMessagesApi/GetRouteTripInfo/$feedMessageId');

  /// Defining the headers required for the API request
  final headers = {'Content-Type': 'application/json'};

  /// Making a GET request to the login API with the encoded JSON body and headers
  final response = await http.get(url, headers: headers);

  // Returning the response from the API
  if (response.statusCode == 200) {
    return RouteTripInfoMessage.fromJson(jsonDecode(response.body));
  } else {
    throw Exception("Failed to load route trip info message");
  }
}
