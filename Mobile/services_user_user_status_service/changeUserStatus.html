Future<http.Response> changeUserStatus(String userId, bool status) async {
  /// Defining the URL for the login API
  final url = Uri.parse('https://traveller-server.azurewebsites.net/api/UserStatusApi/PutUserStatus/${userId}');

  /// Defining the headers required for the API request
  final headers = {'Content-Type': 'application/json'};

  final jsonBody = json.encode({'chageStatus': status});

  /// Making a POST request to the login API with the encoded JSON body and headers
  final response = await http.post(url, headers: headers, body: jsonBody);

  /// Returning the response from the API
  return response;
}
