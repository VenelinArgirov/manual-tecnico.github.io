Future<http.Response> createAlertVote(AlertVote newAlertVote) async {
  final url =
      Uri.parse('https://traveller-server.azurewebsites.net/api/AlertVotes');
  final headers = {'Content-Type': 'application/json'};
  final response = await http.post(url,
      headers: headers, body: json.encode(newAlertVote.toJson()));

  return response;
}
