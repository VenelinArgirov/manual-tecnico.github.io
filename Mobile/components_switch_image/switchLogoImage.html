Image switchLogoImage(Color themeColor){
  /// Atribute to define the default image.
  String logoImage = 'lib/assets/images/logo.png';
  if (themeColor == const Color(0xff3BE2AB)) {
      logoImage = 'lib/assets/images/logo.png';
      return Image.asset(
        logoImage,
        width: 150,
        height: 150,
        );
  } else if (themeColor == const Color(0xff7E57C2)) {
      logoImage = 'lib/assets/images/logo_dark.png';
      return Image.asset(
        logoImage,
        width: 150,
        height: 150,
        );
  }
  return Image.asset(
        logoImage,
        width: 150,
        height: 150,
        );
}
