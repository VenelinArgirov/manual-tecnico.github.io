Future<http.Response> resetPassword(String code, String newPassword) async {
// Defining the URL for the password reset API with the code parameter
final url = Uri.parse('https://traveller-server.azurewebsites.net/api/AuthApi/ResetPassword/$code');

// Defining the headers required for the API request
final headers = {'Content-Type': 'application/json'};

// Defining the request body for the API request with the new password parameter
final requestBody = {'newPassword': newPassword};

// Encoding the request body into a JSON body
final jsonBody = json.encode(requestBody);

// Making a POST request to the password reset API with the encoded JSON body and headers
final response = await http.post(url, headers: headers, body: jsonBody);

// Returning the response from the API
return response;
}
