Future<http.Response> register(String email, String password, String userName) async {
  final url = Uri.parse('https://traveller-server.azurewebsites.net/api/userapi/Register');
  final headers = {'Content-Type': 'application/json'};
  final jsonBody = json.encode({'email': email, 'password': password, 'userName': userName});
  final response = await http.post(url, headers: headers, body: jsonBody);

  return response;
}
