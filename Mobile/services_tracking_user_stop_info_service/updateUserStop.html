Future<List<FeedMessage>> updateUserStop(String userId, double latitude, double longitude) async {
  final url = Uri.parse('https://traveller-server.azurewebsites.net/api/UserStopInfoesApi/Update');
  final headers = {'Content-Type': 'application/json'};
  UserStopInfoMessage info = UserStopInfoMessage(userId: userId, latitude: latitude, longitude: longitude, timeStamp: DateTime.now());
  final jsonBody = json.encode(info);
  final response = await http.post(url, headers: headers, body: jsonBody);

  if (response.statusCode == 200 && response.body.isNotEmpty) {
          return (json.decode(response.body) as List)
        .map((i) => FeedMessage.fromJson(i))
        .toList();
  }
  return [];
}
