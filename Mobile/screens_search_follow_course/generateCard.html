Widget generateCard(
    String htmlInstructions,
    google_directions.Location origin,
    google_directions.Location destination,
    google_directions.TransitDetails transitDetails,
    num duration) {
  return Card(
    color: const Color(0xffFAFAF9),
    child: Padding(
      padding: const EdgeInsets.all(16),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          //Icon(Icons.),
          FutureBuilder<String>(
            future: getAddress(origin),
            builder: (context, snapshot) {
              if (snapshot.hasData) {
                return Text(
                  snapshot.data!,
                  style: const TextStyle(
                      fontSize: 18, fontWeight: FontWeight.w500),
                  overflow: TextOverflow.ellipsis,
                );
              } else {
                return Text(
                  AppLocalizations.of(context)!.missingStreet,
                  style: TextStyle(fontSize: 12),
                );
              }
            },
          ),
          const SizedBox(height: 8),
          Row(
            mainAxisAlignment: MainAxisAlignment.spaceEvenly,
            children: [
              const SizedBox(width: 16),
              Expanded(
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Text('${transitDetails.numStops.toString()} paragens',
                        style: const TextStyle(fontSize: 16)),
                  ],
                ),
              ),
              Expanded(
                child: Row(
                  mainAxisAlignment: MainAxisAlignment.end,
                  children: [
                    Text(formatedTime(timeInSecond: duration).toString(),
                        style: const TextStyle(
                            fontSize: 18, fontWeight: FontWeight.w500)),
                    const Text(" min", style: TextStyle(fontSize: 14)),
                  ],
                ),
              ),
            ],
          ),
          const SizedBox(height: 8),
          FutureBuilder<String>(
            future: getAddress(destination),
            builder: (context, snapshot) {
              if (snapshot.hasData) {
                return Text(
                  snapshot.data!,
                  style: const TextStyle(
                      fontSize: 18, fontWeight: FontWeight.w500),
                  overflow: TextOverflow.ellipsis,
                );
              } else {
                return const Text(
                  "Error Message",
                  style: TextStyle(fontSize: 12),
                );
              }
            },
          ),
        ],
      ),
    ),
  );
}
