Widget getTransportInfoTile(google_directions.Leg leg) {
  return ListTile(
    title: Column(
      crossAxisAlignment: CrossAxisAlignment.start,
      children: [
        Row(
          mainAxisAlignment: MainAxisAlignment.start,
          children: [
            for (int i = 0; i < leg.steps.length; i++) ...[
              if (leg.steps[i].travelMode == TravelMode.transit) ...[
                if (leg.steps[i].transitDetails!.line.vehicle.type == 'BUS')
                  Container(
                    width: 65,
                    height: 25,
                    decoration: BoxDecoration(
                      borderRadius: BorderRadius.circular(8),
                      color: Color(_getColorFromHex(
                          leg.steps[i].transitDetails!.line.color)),
                    ),
                    child: Row(
                      children: [
                        Row(
                          children: [
                            Image.network(
                              'http:' +
                                  leg.steps[i].transitDetails!.line.vehicle
                                      .icon,
                              color: Colors.black,
                            ),
                            Text(leg.steps[i].transitDetails!.line.shortName),
                          ],
                        ),
                      ],
                    ),
                  )
                else
                  Container(
                    width: 25,
                    height: 25,
                    decoration: BoxDecoration(
                      shape: BoxShape.circle,
                      color: Color(_getColorFromHex(
                          leg.steps[i].transitDetails!.line.color)),
                    ),
                    child: Row(
                      children: [
                        Image.network(
                          'http:' +
                              leg.steps[i].transitDetails!.line.vehicle.icon,
                          color: Colors.black,
                        ),
                      ],
                    ),
                  ),
              ] else if (leg.steps[i].travelMode == TravelMode.walking) ...[
                Icon(
                  Icons.directions_walk,
                  color: Colors.black,
                ),
              ],
              if (i < leg.steps.length - 1) ...[
                Text(
                  ' â€¢ ',
                  style: TextStyle(fontSize: 32),
                ),
              ],
            ],
            Text(
              "${(leg.duration.value / 60).round()}min",
              style: TextStyle(fontSize: 16),
            ),
          ],
        ),
      ],
    ),
  );
}
