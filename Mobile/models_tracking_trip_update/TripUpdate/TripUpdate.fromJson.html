factory TripUpdate.fromJson(Map<String, dynamic> json) {
  List<StopTimeUpdate> stopTimes = [];
  if (json?['stopTimeUpdates'] != null) {
    var stopTimeJson = json?['stopTimeUpdates'] as List;
    stopTimes = stopTimeJson
        .map((stopTimeJSON) => StopTimeUpdate.fromJson(stopTimeJSON))
        .toList();
  }

  return TripUpdate(
    tripUpdateId: json['tripUpdateId'],
    tripDescriptor: TripDescriptor.fromJson(json?['tripDescriptor']),
    stopTimeUpdates: stopTimes,
    timeStamp: DateTime.tryParse(json?['timeStamp']),
  );
}
