Future<http.Response> getUserInfoWithEmailFromAPI(String email) async {
// Defining the URL for the users API with the email parameter
  final url = Uri.parse('https://traveller-server.azurewebsites.net/api/UserApi/email/$email');

// Defining the headers required for the API request
  final headers = {'Content-Type': 'application/json'};

// Making a GET request to the back-end API asking for the user info with the specified email
  final response = await http.get(url, headers: headers);

// Returning the response from the API
  return response;
}
