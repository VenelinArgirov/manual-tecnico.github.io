Future<Trip?> getTripFromGoogleInfo(
  String busNumber,
  String headSign,
  String departureStopName,
  DateTime departureStopTime,
  String arrivalStopName,
  DateTime arrivalStopTime)
async {
  final url = Uri.parse('https://traveller-server.azurewebsites.net/api/TripsApi/GetGTFSTrip');
  final headers = {'Content-Type': 'application/json'};
  GoogleTripInfoMessage info = GoogleTripInfoMessage(busNumber: busNumber, headSign: headSign, departureStopName: departureStopName, departureStopTime: departureStopTime, arrivalStopName: arrivalStopName, arrivalStopTime: arrivalStopTime);
  final jsonBody = json.encode(info);
  final response = await http.post(url, headers: headers, body: jsonBody);

  if (response.statusCode == 200) {
    return Trip.fromJson(jsonDecode(response.body));
  } else {
    return null;
  }
}
