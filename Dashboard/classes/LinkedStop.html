import { Route } from "./route";
import { Stop } from "./stop";

/**
 * Class that represents a linked stop.
 * It links a stop and route together.
 */
export class LinkedStop {
  /**
   * Id of the linked stop.
   */
  id: string;
  /**
   * Id of the stop.
   */
  stopId: string;
  /**
   * Id of the route.
   */
  routeId: string;
  /**
   * The time of departure from the stop.
   */
  time: Date;
  /**
   * Whether the linked stop is used in workdays or weekends/holidays.
   */
  day: number;

  /**
   * Constructor for the class.
   */
  constructor(id: string, stopId: string, routeId: string, time: Date, day: number, public stop?: Stop, public route?: Route) {
    this.id = id;
    this.stopId = stopId;
    this.routeId = routeId;
    this.time = new Date(time);
    this.day = day;
    if (stop) this.stop = stop;
    if (route) this.route = route;
  }

  /**
   * Function used to get the hours and minutes from the time variable and put it on a string.
   * */
  getHoursAndMinutesString() {
    return String(this.time.getHours()).padStart(2, '0') + ":" + String(this.time.getMinutes()).padStart(2, '0');
  }
}
