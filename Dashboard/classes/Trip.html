import { TravelDirection } from "../enums/travel-direction";
import { Route } from "./route";

/**
 * Class that represents a Trip.
 */
export class Trip {
  /**
   * Id of the trip.
   */
  tripId: string;
  /**
   * The text normally displayed on a transport about the route, usually its name.
   */
  tripHeadsign: string;
  /**
   * Id of the route.
   */
  routeId: string;
  /**
   * The route.
   */
  route!: Route;
  /**
   * Id of the service.
   */
  serviceId: string;
  /**
   * The direction of the trip.
   */
  direction: TravelDirection;
  /**
   * Id of the shape.
   */
  shapeId: string;

  /**
   * Constructor for the class.
   */
  constructor(tripId: string, tripHeadsign: string, routeId: string,
    serviceId: string, direction: TravelDirection, shapeId: string);

  /**
   * Constructor for the class.
   */
  constructor(tripId: string, tripHeadsign: string, routeId: string,
    serviceId: string, direction: TravelDirection, shapeId: string, route?: Route) {
    this.tripId = tripId;
    this.tripHeadsign = tripHeadsign;
    this.routeId = routeId;
    this.serviceId = serviceId;
    this.direction = direction;
    this.shapeId = shapeId;

    if (route) this.route = route;
  }
}
