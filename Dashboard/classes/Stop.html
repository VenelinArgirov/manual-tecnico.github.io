export class Stop {
  /**
   * Id of the stop.
   */
  stopId: string;
  /**
   * Short text or a number that identifies the stop for riders.
   * The stop_code may be the same as stop_id if it is public facing.
   */
  stopCode: string;
  /**
   * Name of the stop.
   */
  stopName: string;
  /**
   * Latitude of the stop.
   */
  stopLat: number;
  /**
   * Longitude of the stop.
   */
  stopLon: number;

  /**
   * Constructor for the class.
   */
  constructor(stopId: string, stopCode: string, stopName: string, stopLat: number, stopLon: number) {
    this.stopId = stopId;
    this.stopCode = stopCode;
    this.stopName = stopName;
    this.stopLat = stopLat;
    this.stopLon = stopLon;
  }
}
