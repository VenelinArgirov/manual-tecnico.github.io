import { Agency } from "./agency";
import { StopTime } from "./stop-time";

/**
 * Class that represents a Route.
 */
export class Route {
  /**
   * Id of the route.
   */
  routeId: string;
  /**
   * Id of the agency.
   */
  agencyId: string;
  /**
   * The agency.
   */
  agency!: Agency;
  /**
   * The short name of the route, usually the route number.
   */
  routeShortName: string;
  /**
   * The long name of the route, saying the origin and destination.
   */
  routeLongName: string;
  /**
   * The type of the transport of the route
   */
  routeTransportType: number;
  /**
   * The amount of time the route takes from origin to destination.
   */
  routeDuration: number;
  /**
   * The stops belonging to the route
   */
  stops!: StopTime[];

  /**
   * Constructor for the class.
   */
  constructor(routeId: string, agencyId: string, routeShortName: string,
    routeLongName: string, routeTransportType: number, agency?: Agency, stops?: StopTime[]) {
    this.routeId = routeId;
    this.agencyId = agencyId;
    this.routeShortName = routeShortName;
    this.routeLongName = routeLongName;
    this.routeTransportType = routeTransportType;

    if (agency) this.agency = agency;
    if (stops) this.stops = stops;

    this.routeDuration = this.calculateRouteDuration();
  }

  /**
   * Method to calculate the duration of the route based on the stops it has.
   */
  calculateRouteDuration() {
    if (this.stops && this.stops.length != 0) {
      var finishTime = new Date(this.stops[this.stops.length - 1].arrivalTime).getTime();
      var startTime = new Date(this.stops[0].arrivalTime).getTime();
      return Math.floor((finishTime - startTime) / 60000);
    }
    return 0;
  }
}
