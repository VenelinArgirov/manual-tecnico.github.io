import { Stop } from "./stop";
import { Trip } from "./trip";

/**
 * Class that represents a StopTime.
 */
export class StopTime {
  /**
   * Id of the Trip.
   */
  tripId: string;
  /**
   * The trip.
   */
  trip!: Trip;
  /**
   * Arrival time at the stop.
   */
  arrivalTime: Date;
  /**
   * Departure time from the stop
   */
  departureTime: Date;
  /**
   * Id of the stop.
   */
  stopId: string;
  /**
   * The stop.
   */
  stop!: Stop;
  /**
   * Order of the stop in the stopTime.
   */
  stopSequence: number;
  /**
   * Actual distance traveled along the associated shape, from the first stop to the stop specified in this record.
   */
  shapeDistTraveled: number;


  /**
   * Constructor for the class.
   */
  constructor(tripId: string, arrivalTime: Date, departureTime: Date,
    stopId: string, stopSequence: number, shapeDistTraveled: number, trip?: Trip, stop?: Stop) {
    this.tripId = tripId;
    this.arrivalTime = new Date(arrivalTime);
    this.departureTime = new Date(departureTime);
    this.stopId = stopId;
    this.stopSequence = stopSequence;
    this.shapeDistTraveled = shapeDistTraveled;

    if (trip) this.trip = trip;
    if (stop) this.stop = stop;
  }

  /**
   * Function used to get the hours and minutes from the time variable and put it on a string.
   * */
  public getHoursAndMinutesString() {
    return String(this.departureTime.getHours()).padStart(2, '0') + ":" + String(this.departureTime.getMinutes()).padStart(2, '0');
  }
}
