import { Component, AfterViewInit, Injectable, OnInit } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Chart, registerables } from 'chart.js';
import { TranslocoService } from '@ngneat/transloco';
import { Subscription } from 'rxjs';
import { StatisticsService } from '../../services/statistics.service';
import { BreakpointObserver, Breakpoints } from '@angular/cdk/layout';

import { NgModule } from '@angular/core';
import { HttpClientModule } from '@angular/common/http';

@Component({
  selector: 'app-statistics',
  templateUrl: './statistics.component.html',
  styleUrls: ['./statistics.component.css']
})

/**
 * The component is marked as injectable to allow it to be used as a service
 */

@Injectable({
  providedIn: 'root'
})

/**
 * StatisticsComponent class is the main component that implements the OnInit interface
 */

export class StatisticsComponent implements OnInit {
  /**
   * Variables to store the number of active users
   */
  activeUsers?: number;

  /**
   * Variables to store the number of total users
   */
  totalUsers?: number;

  /**
   * Variables to store the number of new users
   */
  newUsers?: number;

  /**
  * The chart variable is used to store the chart object created using the Chart.js library
  */
  chart: any = [];

  /**
  * The label variable is used to store the chart title label
  */
  label: any;

  /**
  * The labelSub variable is used to subscribe to the translated label and update the chart
  */
  labelSub!: Subscription;

  /**
  * Variables to store the most active zones
  */  
  mostActiveZones: string[] = [];

  /**
  * Variables to store the orginal keys of the most active zones
  */  
  mostActiveZonesKeys: string[] = []; 

  /**
  * Variables to store the most active zones data
  */
  mostActiveZonesData: number[] = []; 

  /**
  * Variables responsible to change the number of mat-card elements to show
  */
  change: number = 1;

  /**
 * The constructor injects the HttpClient and TranslocoService modules to be used in the component
 * @param {HttpClient} http HttpClient that can send Http requests, used to communicate with the server.
 * @param {TranslocoService} service TranslocoService object in order to change chart label language at runtime.
 */
  constructor(private breakpointObserver: BreakpointObserver,private statisticsService: StatisticsService, private http: HttpClient, private translation: TranslocoService) {
    Chart.register(...registerables);
    this.breakpointObserver.observe([
      Breakpoints.Small,
      Breakpoints.XSmall,
    ]).subscribe(result => {
      if (result.matches)
        this.change = 3;
      else {
        this.change = 1;
      }
    });
  }

  /**
 * ngOnInit method is called when the component is initialized.
 * It retrieves the total and new users data from the API and creates the chart using Chart.js.
 * It also subscribes to the translated label and updates the chart accordingly.
 */
  ngOnInit(): void {

    this.statisticsService.getActiveUsers().subscribe(data => {
      this.activeUsers = data.activeUsersCount;
    });

    this.statisticsService.getTotalNumberOfUsers().subscribe(data => {
      this.totalUsers = data.totalUsers;
    });

    this.statisticsService.getNumberOfNewUsers().subscribe(data => {
      this.newUsers = data.newUsers;
    });

    this.statisticsService.getMostActiveZones().subscribe(data => {
      this.mostActiveZonesKeys = Object.keys(data);
      this.mostActiveZones = Object.keys(data).map(zone => this.translation.translate('statistics.list.' + zone));
      this.mostActiveZonesData = Object.values(data);
      if (this.chart.data.datasets[0].data.length !== this.mostActiveZonesData.length) {
        this.chart.data.datasets[0].data = this.mostActiveZonesData;
        this.chart.data.labels = this.mostActiveZones;
        this.chart.update();
      }
      this.chart.update();
    });

    this.label = this.translation.selectTranslate('statistics.list.chart-title');
    this.labelSub = this.translation.selectTranslate('statistics.list.chart-title').subscribe(
      label => {
        this.label = label;
        this.chart.data.datasets[0].label = this.label;
        this.chart.update();
      }
    );

    this.translation.langChanges$.subscribe(() => {
      this.mostActiveZones = this.mostActiveZonesKeys.map(zone => this.translation.translate('statistics.list.' + zone));
      this.chart.data.labels = this.mostActiveZones;
      this.chart.update();
    });

    const canvas = <HTMLCanvasElement>document.getElementById('myChart');
    const ctx = canvas.getContext('2d');

    if (ctx) {
      this.chart = new Chart(ctx, {
        type: 'bar',
        data: {
          labels: this.mostActiveZones,
          datasets: [
            {
              label: this.label,
              data: this.mostActiveZonesData, 
              backgroundColor: [
                'rgba(59, 226, 171, 1)',
              ]
            },
          ],

        },

        options: {
          plugins: { 
            legend: {
              labels: {
                color: "white"
              }
            }
          },
        },
      });
    }

  }
}
