import { Component, OnInit, ViewChild } from '@angular/core';
import { User } from '../user';
import { UserService } from '../user.service';
import { SuspensionService } from '../suspension.service';
import { SelectionModel } from '@angular/cdk/collections';
import { MatTableDataSource } from '@angular/material/table';
import { MatPaginator } from '@angular/material/paginator';
import Snackbar from 'awesome-snackbar';
import { TranslocoService } from '@ngneat/transloco';

@Component({
  selector: 'app-users-management',
  templateUrl: './users-management.component.html',
  styleUrls: ['./users-management.component.css']
})
/**
* Class for the component of the page of Users Management
*/
export class UsersManagementComponent implements OnInit {
  /**
   * The paginator used in the users management table.
   */
  @ViewChild(MatPaginator) paginator!: MatPaginator;
  /**
   * All the existent users.
   */
  public users: User[] = [];
  /**
   * Mapping the banned users in the system.
   */
  public banned: Map<string, boolean> = new Map<string, boolean>();
  /**
   * The columns to be displayed in the users management table.
   */
  displayedColumns: string[] = ['select', 'creation-date', 'user-name', 'user-rating', 'user-rank', 'user-details'];
  /**
   * The data source used in the users management table.
   */
  dataSource!: MatTableDataSource<User>
  /**
   * Holds the routes selected via checkbox.
   */
  selection = new SelectionModel<User>(true, []);


/**
 * Constructor for the component.
 * Gets the users in the system and add them to the "users" through the function getUsersInfo().
 * @param userService Service that communicates with the server and get all users and informations.
 * @param suspensionService Service that communicates with the server and has various operations about Suspensions.
 * @param translocoService Service to perform the translation on the snackbar text.
 */
  constructor(private userService: UserService, private suspensionService: SuspensionService, private translocoService: TranslocoService) { }

  /**
   * Component inicializer.
   */
  ngOnInit(): void {
    this.getUsersInfo();
  }

  /**
   *  Whether the number of selected elements matches the total number of rows.
   */
  isAllSelected() {
    const numSelected = this.selection.selected.length;
    const numRows = this.dataSource.data.length;
    return numSelected === numRows;
  }

  /**
   *  Selects all rows if they are not all selected; otherwise clear selection.
   */
  toggleAllRows() {
    if (this.isAllSelected()) {
      this.selection.clear();
      return;
    }
    this.selection.select(...this.dataSource.data);
  }

  /**
   * Function used to verify if an user is banned permant
   */
  getUsersSuspensionPermanent() {
    this.users.forEach((user) => {
      this.suspensionService.getSuspensionsOfUser(user.id).subscribe(res => {
        this.banned.set(user.id, false);
        res.forEach((suspension) => {
          if (suspension.permanent) {
            this.banned.set(user.id, true);
          }
        });
        if (user.userRank.toLowerCase() == 'admin') {
          this.banned.set(user.id, true);
        };
      })
    })
  }

  /**
   * Function used to get the users in the system and add them to the "users" attribute.
   */
  getUsersInfo() {
    this.users = [];
    this.userService.getUsers().subscribe((usr) => {
      usr.forEach((user) => {
        this.users.push(new User(user.id, user.userRank, user.userName, user.rating, user.creationDate, user.email, "null"));
      });
      this.getUsersSuspensionPermanent();
      this.dataSource = new MatTableDataSource<User>(this.users);
      this.dataSource.paginator = this.paginator;
    });
  }

  /**
  * Ban all the checked users.
  */
  banUsers() {
    this.selection.selected.forEach((user) => {
      var userToBanId = user.id;
      this.suspensionService.suspendUser(userToBanId, new Date(), new Date("2099-01-01"), "", true).subscribe(res => {
        new Snackbar(this.translocoService.translate('user-details.user-banned'), {
          position: 'bottom-center',
          actionText: this.translocoService.translate('user-details.hide'),
          style: {
            container: [
              ['background-color', '#003F91']
            ]
          }
        });
        this.getUsersSuspensionPermanent();
      })
    });
  }

  /**
   * Unban an user account with a button.
   * @param userId Id of the user account who is banned.
   */
  unbanUser(userId: string) {
    console.log(userId);
    this.suspensionService.unbanUser(userId).subscribe(res => {
      new Snackbar(this.translocoService.translate('user-details.user-unbanned'), {
        position: 'bottom-center',
        actionText: this.translocoService.translate('user-details.hide'),
        style: {
          container: [
            ['background-color', '#003F91']
          ]
        }
      });
      this.getUsersSuspensionPermanent();
    })
  }

  /**
   * Aplies a filter to the list of existing routes.
   * @param {Event} event The parameter to filter the list for.
   */
  applyFilter(event: Event) {
    const filterValue = (event.target as HTMLInputElement).value;
    this.dataSource.filter = filterValue.trim().toLowerCase();
  }

}
