import { Component, Inject, OnInit, Optional } from '@angular/core';
import { AbstractControl, FormControl, ValidationErrors, ValidatorFn, Validators } from '@angular/forms';
import { ActivatedRoute } from '@angular/router';
import { TranslocoService } from '@ngneat/transloco';
import Snackbar from 'awesome-snackbar';
import { Suspension } from '../../models/suspension';
import { SuspensionService } from '../../services/suspension.service';
import { User } from '../../models/user';
import { UserService } from '../../services/user.service';
import { StatisticsService } from '../../services/statistics.service'

@Component({
  selector: 'app-user-details',
  templateUrl: './user-details.component.html',
  styleUrls: ['./user-details.component.css']
})

/**
 * Class for the component of the page of viewing User details.
 */
export class UserDetailsComponent implements OnInit {
  /**
   * Current user.
   */
  public currentUser!: User;
  /**
   * All the suspensions of the current user.
   */
  public suspensions!: Suspension[];
  /**
   * Whether the user is banned.
   */
  public banned!: boolean;
  /**
   * Reason for the user ban.
   */
  public banReason!: string;
  /**
   * Date until the suspension of the user is over.
   */
  public suspendDate = new FormControl('', [Validators.required, this.dateAfterTodayValidator()]);
  /**
   * Reason for the suspension.
   */
  public suspendReason = "";
  /**
   * Most used transport type.
   */
  public mostUsedTransportType!: string;
  /**
   * Time spent in transport.
   */
  public timeSpentInTransport?: string;
  /**
   * Constructor for the component.
   * @param {ActivatedRoute} route Gives information about the current route.
   * @param {UserService} userService Service that communicates with the server and has various operations about Users.
   * @param {SuspensionService} suspensionService Service that communicates with the server and has various operations about Suspensions.
   */
  constructor(private route: ActivatedRoute, private userService: UserService, private suspensionService: SuspensionService, private translocoService: TranslocoService, private statisticsService: StatisticsService) {
  }

  /**
   * Gets the user through the id in the url.
   */
  getUser() {
    this.userService.getUser(this.route.snapshot.params.id).subscribe(result => {
      this.currentUser = result;
    }, error => console.error(error))
  }

  /**
   * Gets the suspensions of the user whose Id is in the url.
   * Sorts the suspensions in descending order and checks if any suspension is permanent.
   */
  getSuspensions() {
    this.suspensionService.getSuspensionsOfUser(this.route.snapshot.params.id).subscribe(result => {
      this.suspensions = result;
      this.suspensions.sort((a: Suspension, b: Suspension) => {
        if (a.startDate > b.startDate) {
          return 1;
        } else return 0;
      })
      this.suspensions.forEach((suspension) => {
        if (suspension.permanent == true) {
          this.banned = true;
          this.banReason = suspension.reason;
        }
      })
    }, error => console.error(error))
  }

  /**
   * Suspends a user for a interval of time obtained through the form.
   * @param {boolean} permanent Whether or not the suspension is permanent.
   */
  suspendUser(permanent: boolean) {
    if (!permanent) {
      this.suspensionService.suspendUser(this.currentUser.id, new Date(), new Date(this.suspendDate.value as string), this.suspendReason, permanent).subscribe(res => {
        new Snackbar(this.translocoService.translate('user-details.user-suspended'), {
          position: 'bottom-center',
          actionText: this.translocoService.translate('user-details.hide'),
          style: {
            container: [
              ['background-color', '#003F91']
            ]
            }
          });
        this.resetForms();
        this.ngOnInit();
      })
    } else {
      this.suspensionService.suspendUser(this.currentUser.id, new Date(), new Date("2099-01-01"), this.suspendReason, permanent).subscribe(res => {
        new Snackbar(this.translocoService.translate('user-details.user-banned'), {
          position: 'bottom-center',
          actionText: this.translocoService.translate('user-details.hide'),
          style: {
            container: [
              ['background-color', '#003F91']
            ]
          }
        });
        this.resetForms();
        this.ngOnInit();
      })
    }
  }

  /**
   * Function to unban an user.
   */
  unbanUser() {
    this.suspensionService.unbanUser(this.currentUser.id).subscribe(res => {
      this.resetForms();
      this.ngOnInit();
      new Snackbar(this.translocoService.translate('user-details.user-unbanned'), {
        position: 'bottom-center',
        actionText: this.translocoService.translate('user-details.hide'),
        style: {
          container: [
            ['background-color', '#003F91']
          ]
        }
      });
    })
  }

  /**
   * Returns the possible error messages related to the user input while trying to suspend a user, specifically the date input.
   * Checking if either there isn't a valid inserted or if the inserted date is before today
   * */
  getErrorMessage() {
    if (this.suspendDate.value as string == "") {
      return 'no-date';
    } else if (new Date(this.suspendDate.value as string) < new Date()) {
      return 'before-date';
    }
    return '';
  }

  /**
   * Gets the user most used transport type.
   */
  getUserMostUsedTransport() { 
    this.statisticsService.getUserMostUsedTransport(this.route.snapshot.params.id).subscribe(data => {
      this.mostUsedTransportType = data.mostUsedTransport;
    });
  }


  /**
   * Gets the user total time spent in transportation.
   */
  getUserTimeSpent() {
    this.statisticsService.getUserTimeSpent(this.route.snapshot.params.id).subscribe(data => {
      this.timeSpentInTransport = data.timeSpent;
    });
  }

  /**
   * Resets the form data.
   */
  resetForms() {
    this.suspendReason = "";
    this.suspendDate.reset();
  }

  ngOnInit(): void {
    this.getUser();
    this.getSuspensions();
    this.getUserMostUsedTransport();
    this.getUserTimeSpent();
  }

  /**
   * Custom validator created for the input suspension date.
   * Checking if the date inserted by the user is before today or not.
   * */
  dateAfterTodayValidator(): ValidatorFn {
    return (control: AbstractControl): ValidationErrors | null => {
      const value = new Date(control.value as string);
      if (!value) {
        return null;
      }
      const beforeToday = value < new Date();
      return beforeToday ? { beforeToday: true } : null;
    }
  }

}
