import { Component, OnInit } from '@angular/core';
import { RouteService } from '../../services/route.service';
import { AbstractControl, FormArray, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { Route } from '../../models/route';
import { Stop } from '../../models/stop';
import { StopService } from '../../services/stop.service';
import { StopTime } from "../../models/stop-time";
import { Guid } from 'guid-typescript';
import { map, Observable } from 'rxjs';
import { MapDirectionsService } from '@angular/google-maps';
import { ActivatedRoute, Router } from '@angular/router';
import { TravelDirection } from '../../enums/travel-direction';
import { Trip } from '../../models/trip';
import { TripService } from '../../services/trip.service';
import { StopTimeService } from '../../services/stop-time.service';
import Snackbar from 'awesome-snackbar';
import { TranslocoService } from '@ngneat/transloco';

@Component({
  selector: 'app-trip-create',
  templateUrl: './trip-create.component.html',
  styleUrls: ['./trip-create.component.css']
})
/**
 * Class for the component of the page of Route Editing
 */
export class TripCreateComponent implements OnInit {
  /**
   * Options to be used by the Google-Map field.
   */
  mapOptions: google.maps.MapOptions = {
    mapTypeId: 'roadmap',
    disableDefaultUI: true,
    zoomControl: true,
    clickableIcons: false,
    disableDoubleClickZoom: true,
    draggable: false,
  };
  /**
   * Options to be used by the Directions-Renderer field.
   */
  directionsOptions: google.maps.DirectionsRendererOptions = {
    suppressMarkers: true,
    suppressInfoWindows: true,
  };
  /**
   * Options to be used by the Map-Marker field.
   */
  markerOptions: google.maps.MarkerOptions = { draggable: false };
  /**
   * Zoom level for the map to use at the start.
   */
  zoom = 11;
  /**
   * Coordinates for the map to be centered on.
   */
  center!: google.maps.LatLngLiteral;
  /**
   * List of markers that represents the stops of the route to edit.
   */
  markers: Array<google.maps.LatLng> = new Array<google.maps.LatLng>();
  /**
   * Holds the result from the call of the Directions API made from the list of markers.
   */
  directionsResults$!: Observable<google.maps.DirectionsResult | undefined>;

  /**
 * Id of the route.
 */
  routeId: string = '';
  /**
   * Route to edit.
   */
  route!: Route;
  /**
   * TransportType for reverse mapping in the html.
   */
  TravelDirection = TravelDirection;
  /**
   * All the existent stops.
   */
  AllStops: Stop[] = [];

  /**
 * All the existent trips.
 */
  AllTrips: Trip[] = [];
  /**
   * All the values of the TransportType enum.
   */
  directions: number[] = Object.keys(TravelDirection).map(v => Number(v)).filter(v => Number.isFinite(v));

  /**
   * Form data to be viewed in the page
   */
  createRouteForm = this.formBuilder.group({
    tripName: ['', Validators.required],
    tripTransportDirection: this.formBuilder.control<number | null>(null, Validators.required),
    stops: this.formBuilder.array([]),
  });

  /**
   * Constructor for the component.
   * Gets the route whose Id matches the one in the URL and passes its info to the form.
   * Also gets all the existent stops.
   * @param {ActivatedRoute} actRoute Gives information about the current route.
   * @param {RouteService} routeService Service that communicates with the server and has various operations about Routes.
   * @param {StopService} stopService Service that communicates with the server and has various operations about Stops.
   * @param {FormBuilder} formBuilder Builds form data.
   * @param {MapDirectionsService} mapDirectionsService Google service of the Directions API.
   * @param {TripService} tripService Service that communicates with the server and has various operations about Trips.
   * @param {Router} router Router.
   * @param {StopTimeService} stopTimeService Service that communicates with the server and has various operations about StopTimes.
   * @param {TranslocoService} translocoService Service that perform the translation of the text.
   */
  constructor(private urlRoute: ActivatedRoute, private routeService: RouteService, private stopService: StopService,
    private formBuilder: FormBuilder, private mapDirectionsService: MapDirectionsService, private tripService: TripService,
    private router: Router, private stopTimeService: StopTimeService, private translocoService: TranslocoService) { }

  ngOnInit(): void {
    this.routeId = String(this.urlRoute.snapshot.paramMap.get("id"));
    this.routeService.getRoute(this.routeId)
      .subscribe((route) => {
        this.stopTimeService.getStopsOfRoute(route.routeId).subscribe((stops) => {
          this.route = new Route(route.routeId, route.agencyId, route.routeShortName, route.routeLongName,
            route.routeTransportType, route.agency, stops);
        });
      });

    this.tripService.getTripsOfRoute(this.routeId).subscribe((trips) => {
      trips.forEach((trip) => {
        this.AllTrips!.push(new Trip(trip.tripId, trip.tripHeadsign, trip.routeId, trip.serviceId, trip.direction, trip.shapeId));
      });
      
    });


    this.stopService.getStops().subscribe(result => {
      this.AllStops = result;
    }, error => console.error(error));
  }

  /**
   * Adds a form to the page that lets the user select the details for a stop in the route.
   */
  addStop() {
    const stopForm = this.formBuilder.group({
      stopId: '',
      stopTime: ''
    });

    this.stops.push(stopForm);
  }

  /**
   * Removes the form of one of the route stops.
   * @param {number} stopIndex The form index to delete.
   */
  removeStop(stopIndex: number) {
    this.stops.removeAt(stopIndex);
    this.updateMap();
  }

  /**
   * Gets all the stop forms in the page.
   */
  get stops() {
    return this.createRouteForm.controls["stops"] as FormArray;
  }

  /**
   * Casts a AbstractControl form to a FormGroup.
   */
  toFormGroup = (form: AbstractControl) => form as FormGroup;

  /**
   * Updates the map in the page by adding/removing markers and linking them together in one route.
   */
  updateMap() {
    this.addMarkers();
    this.linkMarkers();
  }

  /**
   * Iterates through the list of stop forms and creates a new marker from the corresponding stop and its coordinates.
   */
  private addMarkers() {
    this.markers.splice(0);

    this.createRouteForm.value.stops?.forEach(s => {
      var stop = (s as formStop);
      this.AllStops.forEach(compStop => {
        if (compStop.stopId == stop.stopId) this.markers.push(new google.maps.LatLng(compStop.stopLat, compStop.stopLon));
      });
    });
    this.center = {
      lat: this.markers[Math.floor(this.markers.length / 2)].lat(),
      lng: this.markers[Math.floor(this.markers.length / 2)].lng()
    }
  }

  /**
   * Using the Google Directions API, shows the path the route takes using the first and last marker in the route.
   */
  private linkMarkers() {
    const request: google.maps.DirectionsRequest = {
      origin: this.markers[0],
      destination: this.markers[this.markers.length - 1],
      travelMode: google.maps.TravelMode.DRIVING,
    };
    this.directionsResults$ = this.mapDirectionsService.route(request).pipe(map(response => response.result));
  }

  /**
   * Creates a trip object and sends it to the database using the tripService
   */
  onSubmit(): void {
    if (this.createRouteForm.value.stops != null) {
      if (this.createRouteForm.value.stops.length > 2) {
        const stops = this.createRouteForm.value.stops as formStop[];
        if (stops.every(s => s.stopId.length != 0 && s.stopTime.length != 0)) {
          const tripId = Guid.create().toString();
          let trip: Trip = new Trip(tripId, this.createRouteForm.controls.tripName.value!, this.routeId,
            "1", Number(this.createRouteForm.controls.tripTransportDirection.value!), "4560_0_1_a")

          this.tripService.addTrip(trip).subscribe(() => { }, () => { }, () => {
            let idx = 1;
            this.createRouteForm.value.stops!.forEach(s => {
              var stop = (s as formStop);
              var date = new Date();
              date.setUTCHours(Number(stop.stopTime.substring(0, 2)));
              date.setUTCMinutes(Number(stop.stopTime.substring(3, 5)));
              let stopTime: StopTime = new StopTime(tripId, date, date, stop.stopId, idx, 0.0);
              this.stopTimeService.addStopTime(stopTime).subscribe();
              idx++;
            });
            this.router.navigate(['/', 'route-management']);
          });
        }
        else {
          new Snackbar(this.translocoService.translate('route-management.create-edit-trip.error-message-empty-stops'), {
            position: 'bottom-center',
            actionText: this.translocoService.translate('login.list.hide'),
            style: {
              container: [
                ['background-color', '#003F91']
              ]
            }
          });
        }
      } else {
        new Snackbar(this.translocoService.translate('route-management.create-edit-trip.error-message'), {
          position: 'bottom-center',
          actionText: this.translocoService.translate('login.list.hide'),
          style: {
            container: [
              ['background-color', '#003F91']
            ]
          }
        });
      }
    }

    
  }

  /**
   * Cancels the creation of a new stop. Redirects back to the previous page.
   */
  cancel(): void {
    window.history.back();
  }
}

/**
 * Interface for the stop form
 */
interface formStop {
  stopId: string,
  stopTime: string
}
