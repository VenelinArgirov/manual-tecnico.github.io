import { Component, OnDestroy, OnInit } from '@angular/core';
import { RouteService } from '../../services/route.service';
import { ActivatedRoute, convertToParamMap } from '@angular/router';
import { TransportType } from '../../enums/transport-type';
import { StopService } from '../../services/stop.service';
import { MapDirectionsService } from '@angular/google-maps';
import { Router } from '@angular/router';
import { Agency } from '../../models/agency';
import { AgencyService } from '../../services/agency.service';
import { AbstractControl, FormArray, FormBuilder, FormGroup, Validators } from '@angular/forms';
import { TripService } from '../../services/trip.service';
import { StopTimeService } from '../../services/stop-time.service';
import { Route } from '../../models/route';

@Component({
  selector: 'app-route-edit',
  templateUrl: './route-edit.component.html',
  styleUrls: ['./route-edit.component.css']
})
/**
 * Class for the component of the page of Route Editing.
 */
export class RouteEditComponent implements OnInit {
  /**
    * TransportType for reverse mapping in the html.
    */
  TransportType = TransportType;
  /**
   * All the values of the TransportType enum.
   */
  transports: number[] = Object.keys(TransportType).map(v => Number(v)).filter(v => Number.isFinite(v));
  /**
   * List of all the existent agencies.
   */
  public agencies!: Array<Agency>;

  /**
   * Form data to be viewed in the page.
   */
  editRouteForm = this.formBuilder.group({
    number: ['', Validators.required],
    name: ['', Validators.required],
    transportType: this.formBuilder.control<number | null>(null, Validators.required),
    agency: this.formBuilder.control<string | null>(null, Validators.required),
  });

  /**
 * Route whose details are viewed.
 */
  public route!: Route;


  /**
   * Constructor for the component.
   * Gets the route whose Id matches the one in the URL and passes its info to the form.
   * Also gets all the existent stops.
   * @param {ActivatedRoute} actRoute Gives information about the current route.
   * @param {RouteService} routeService Service that communicates with the server and has various operations about Routes.
   * @param {FormBuilder} formBuilder Builds form data.
   * @param {Router} router Router.
   * @param {AgencyService} agencyService Service that communicates with the server and has various operations about Agencies.
   * @param {StopTimeService} stopTimeService Service that communicates with the server and has various operations about StopTimes.
   */
  constructor(private urlRoute: ActivatedRoute, private routeService: RouteService, private formBuilder: FormBuilder, private router: Router, private agencyService: AgencyService) { }

  ngOnInit(): void {
    const id = String(this.urlRoute.snapshot.paramMap.get("id"));
    this.routeService.getRoute(id)
      .subscribe((route) => {
        this.route = new Route(route.routeId, route.agencyId, route.routeShortName, route.routeLongName, route.routeTransportType, route.agency);

          this.editRouteForm.controls.number.patchValue(this.route.routeShortName);
          this.editRouteForm.controls.name.patchValue(this.route.routeLongName);
          this.editRouteForm.controls.transportType.patchValue(this.route.routeTransportType);
          this.editRouteForm.controls.agency.patchValue(this.route.agencyId);

      });
    this.agencyService.getAgencies().subscribe((agencies) => { this.agencies = agencies; });
  }

  ///**
  // * Creates a route object and sends it to the database using the routeService.
  // */
  onSubmit(): void {
    let route: Route = new Route(this.route.routeId, this.editRouteForm.controls.agency.value!, this.editRouteForm.controls.number.value!, this.editRouteForm.controls.name.value!,
      this.editRouteForm.controls.transportType.value!);
    this.routeService.editRoute(route).subscribe();
    this.router.navigate(['/', 'route-details', this.route.routeId]);
  }

  /**
   * Cancels the creation of a new stop. Redirects back to the previous page.
   */
  cancel(): void {
    window.history.back();
  }
}
