import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormControl, Validators } from '@angular/forms';
import { TranslocoService } from '@ngneat/transloco';
import Snackbar from 'awesome-snackbar';
import { Guid } from 'guid-typescript';
import { Stop } from '../../models/stop';
import { StopService } from '../../services/stop.service';

@Component({
  selector: 'app-stop-create',
  templateUrl: './stop-create.component.html',
  styleUrls: ['./stop-create.component.css']
})

/**
 * Class for the component of the page for creating Stops
 */
export class StopCreateComponent implements OnInit {
  /**
   * Options to be used by the Google-Map field
   */
  options: google.maps.MapOptions = {
    mapTypeId: 'roadmap',
    disableDefaultUI: true,
    zoomControl: true,
    clickableIcons: false,
    disableDoubleClickZoom: true,
  };
  /**
   * Options to be used by the Map-Marker field
   */
  markerOptions: google.maps.MarkerOptions = { draggable: false };

  /**
   * Zoom level for the map to use at the start
   */
  zoom = 13;
  /**
   * Coordinates for the map to be centered on
   */
  center!: google.maps.LatLngLiteral;
  /**
   * Coordinates of the stop marker
   */
  markerPos!: google.maps.LatLngLiteral | undefined;

  /**
   * Form data to be viewed in the page
   */
  createStopForm = this.formBuilder.group({
    code: ['', Validators.required],
    name: ['', Validators.required],
    latitude: this.formBuilder.control<number | null>(null, Validators.required),
    longitude: this.formBuilder.control<number | null>(null, Validators.required)
  });

  /**
   * Constructor for the component.
   * @param {StopService} stopService Service that communicates with the server and has various operations about Stops.
   * @param {FormBuilder} formBuilder Builds form data.
   * @param {TranslocoService} translocoService Service that perform the translation of the text.
   */
  constructor(private stopService: StopService, private formBuilder: FormBuilder, private translocoService: TranslocoService) { }

  ngOnInit(): void {
    navigator.geolocation.getCurrentPosition((position) => {
      this.center = {
        lat: position.coords.latitude,
        lng: position.coords.longitude,
      };
    });
  }

  /**
   * Adds a map marker to the map and puts its coordinates in the creation form.
   * @param {google.maps.MapMouseEvent} event Event that contains the coordinates of where the user clicked.
   */
  addStopMarkerInMap(event: google.maps.MapMouseEvent) {
    this.markerPos = event.latLng?.toJSON();
    if (event.latLng?.lat() && event.latLng?.lng()) {
      this.createStopForm.controls.latitude.patchValue(event.latLng.lat());
      this.createStopForm.controls.longitude.patchValue(event.latLng.lng());
    }
  }

  /**
   * Creates a Location object, then sends it to the database using the locationService.
   * After this using the locationId of this object, creates a Stop object and using the stopService sends it to the database.
   */
  onSubmit(): void {
    let stopCode = this.createStopForm.controls.code.value!;
    let stopName = this.createStopForm.controls.name.value!;
    let stopLat = this.createStopForm.controls.latitude.value!;
    let stopLon = this.createStopForm.controls.longitude.value!;

    if (stopCode && stopName && stopLat && stopLon) {

      let stop: Stop = new Stop(Guid.create().toString(), stopCode, stopName, stopLat, stopLon);
      this.stopService.addStop(stop).subscribe();
      new Snackbar(this.translocoService.translate('route-management.add-stop.success-message'), {
        position: 'bottom-center',
        actionText: this.translocoService.translate('login.list.hide'),
        style: {
          container: [
            ['background-color', '#003F91']
          ]
        }
      });
      window.history.back();
    }
    else {
      new Snackbar(this.translocoService.translate('route-management.add-stop.error-message'), {
        position: 'bottom-center',
        actionText: this.translocoService.translate('login.list.hide'),
        style: {
          container: [
            ['background-color', '#003F91']
          ]
        }
      });
    }
  }

  /**
   * Cancels the creation of a new stop. Redirects back to the previous page.
   */
  cancel(): void {
    window.history.back();
  }
}


<ng-container *transloco="let t; read: 'route-management.add-stop'">

  <form [formGroup]="createStopForm" (ngSubmit)="onSubmit()">

          <mat-label>{{ t('label-latitude') }}</mat-label>

          <mat-error *ngIf="!createStopForm.value.latitude">{{ t('error-latitude') }}</mat-error>
        </mat-form-field>

        </map-marker>
      </google-map>
    </div>
  </form>
</ng-container>
