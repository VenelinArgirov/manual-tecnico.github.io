import { HttpClient, HttpErrorResponse } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { Guid } from 'guid-typescript';
import { catchError, Observable, throwError } from 'rxjs';
import { Statistics } from '../models/statistics';

interface MostUsedTransportResponse {
  mostUsedTransport: { transportType: string, totalCount: number }[]
}

@Injectable({
  providedIn: 'root'
})


/**
* Service that communicates with the server and has various operations about statistics.
*/
export class StatisticsService {

  /**
   * Constructor for the class.
   * @param http
   */
  constructor(private http: HttpClient) { }

  /**
  * getActiveUsers method is used to get the total number of active users from the API
  */

  getActiveUsers() {
    //return this.http.get<Statistics>('api/StatisticsApi/GetActiveUsers')
    return this.http.get<Statistics>('https://traveller-server.azurewebsites.net/api/StatisticsApi/GetActiveUsers')
      .pipe(catchError(this.handleError));
  }


  /**
   * getTotalNumberOfUsers method is used to get the total number of users from the API
   */

  getTotalNumberOfUsers() {
    //return this.http.get<Statistics>('api/StatisticsApi/GetTotalNumberOfUsers')
    return this.http.get<Statistics>('https://traveller-server.azurewebsites.net/api/StatisticsApi/GetTotalNumberOfUsers')
      .pipe(catchError(this.handleError));
  }

  /**
  * getNumberOfNewUsers method is used to get the number of new users from the API
  */

  getNumberOfNewUsers() {
    //return this.http.get<Statistics>('api/StatisticsApi/GetTotalOfNewUsers')
    return this.http.get<Statistics>('https://traveller-server.azurewebsites.net/api/StatisticsApi/GetTotalOfNewUsers')
      .pipe(catchError(this.handleError));
  }

  /**
  * getMostActiveZones method is used to get the most active zones from the API
  */

  getMostActiveZones() {
    //return this.http.get<Statistics>('api/StatisticsApi/GetMostActiveZones')
    return this.http.get<Statistics>('https://traveller-server.azurewebsites.net/api/StatisticsApi/GetMostActiveZones')
      .pipe(catchError(this.handleError));
  }

  /**
  * getUserMostUsedTransport method is used to get the most used transport
  */

  getUserMostUsedTransport(userId: string) {
    //return this.http.get<Statistics>(`api/StatisticsApi/GetMostUsedTransport/${userId}`)
    return this.http.get<Statistics>(`https://traveller-server.azurewebsites.net/api/StatisticsApi/GetMostUsedTransport/${userId}`)
      .pipe(catchError(this.handleError));
  }

  /**
  * getUserTimeSpent method is used to get the user time spent in transports
  */

  getUserTimeSpent(userId: string) {
    //return this.http.get<Statistics>(`api/StatisticsApi/GetTimeSpent/${userId}`)
    return this.http.get<Statistics>(`https://traveller-server.azurewebsites.net/api/StatisticsApi/GetTimeSpent/${userId}`)
      .pipe(catchError(this.handleError));
  }

  /**
   * Handles the error if the HttpRequest fails.
   * Prints the error to the console.
   * @param {HttpErrorResponse} error Error to handle.
   */

  private handleError(error: HttpErrorResponse) {
    if (error.status === 0) {
      // A client-side or network error occurred. Handle it accordingly.
      console.error('An error occurred:', error.error);
    } else {
      // The backend returned an unsuccessful response code.
      // The response body may contain clues as to what went wrong.
      console.error(
        `Backend returned code ${error.status}, body was: `, error.error);
    }
    // Return an observable with a user-facing error message.
    return throwError(() => new Error('Something bad happened; please try again later.'));
  }
}
