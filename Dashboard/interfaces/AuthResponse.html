import { Component, OnInit } from '@angular/core';
import { FormControl, FormGroup, Validators } from '@angular/forms';
import { Router, ActivatedRoute } from '@angular/router';
import { AuthService } from '../../services/auth.service';
import { HttpErrorResponse } from '@angular/common/http';
import { TranslocoService } from '@ngneat/transloco';
import Snackbar from 'awesome-snackbar';

@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.css']
})

/**
* Class for the component of the page of viewing Login Page.
*/
export class LoginComponent implements OnInit {
  /**
   * Url to navigate to the statistics page.
   */
  private returnUrl!: string;
  /*
   * Form to have the values of the inputs.
   */
  loginForm!: FormGroup;
  /*
   * Variable used to give feedback to the user if the page is loading or not.
   */
  isLoading!: boolean;
  /**
   * All the available languages of the application.
   */
  private availableLangs: string[] | { id: string, label: string }[];
  /**
   * Current active language.
   */
  private activeLang: string;

  /**
   * Constructor for the component.
   * 
   * @param authService Authentication service that comunicates with the server to validate the inputs of the user.
   * @param router Redirect the current link after the login succeed.
   * @param route Gives information about the current route.
   * @param translocoService Service that perform the translation of the text.
   */
  constructor(private authService: AuthService, private router: Router, private route: ActivatedRoute, private translocoService: TranslocoService)
  {
    this.availableLangs = this.translocoService.getAvailableLangs();
    this.activeLang = this.translocoService.getActiveLang();
  }

  ngOnInit(): void {
    this.loginForm = new FormGroup({
      email: new FormControl(null, [
        Validators.required,
        Validators.email
      ]),
      password: new FormControl(null, [Validators.required]),
    })
    this.returnUrl = this.route.snapshot.queryParams['returnUrl'] || '/';
  }

  /**
   * Function that will be executed when the login button
   * is pressed which serves to send the data to the authentication
   * service if they are correct login was done correctly.
   * 
   * @param loginForm FormGroup to obtain the values of the inputs
   */
  loginCredentials(loginForm: FormGroup) {
    const login = loginForm.value;

    const userForAuth: UserForAuthentication = {
      email: login.email,
      password: login.password
    }

    this.authService.loginUser(userForAuth)
      .subscribe({
        next: (res: AuthResponse) => {
          localStorage.setItem("token", res.token);
          localStorage.setItem("userEmail", login.email);
          this.authService.sendAuthStateChangeNotification(res.isAuthSuccessful);
          this.isLoading = true;
          this.router.navigate(['/statistics']);
        },
        error: (err: HttpErrorResponse) => {
          this.isLoading = false;
          new Snackbar(this.translocoService.translate('login.list.' + err.message), {
            position: 'bottom-center',
            actionText: this.translocoService.translate('login.list.hide'),
            style: {
              container: [
                ['background-color', '#003F91']
              ]
            }
          });
        }
      })
  }

  /**
 * Changes the active language.
 * @param {string} lang id of language to change to.
 */
  changeLang(lang: string) {
    this.translocoService.setActiveLang(lang);
    this.activeLang = lang;
  }
}

export interface UserForAuthentication {
  email: string;
  password: string;
}

export interface AuthResponse {
  isAuthSuccessful: boolean;
  errorMessage: string;
  token: string;
}
