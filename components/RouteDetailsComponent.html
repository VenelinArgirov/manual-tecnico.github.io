import { Component, OnInit } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { LinkedStop } from '../linked-stop';
import { Route } from '../route';
import { RouteService } from '../route.service';

@Component({
  selector: 'app-route-details',
  templateUrl: './route-details.component.html',
  styleUrls: ['./route-details.component.css']
})


/**
 * Class for the component of the page of Route Details
 */
export class RouteDetailsComponent implements OnInit {
  /**
   * @ignore
   */
  helpMessage = `Informação detalhada de uma rota
  Lista de paragens
  Horário`;

  /**
   * Route whose details are viewed.
   */
  public route!: Route;

  /**
   * Constructor for the component.
   * @param {ActivatedRoute} urlRoute Gives information about the current route.
   * @param {RouteService} routeService Service that communicates with the server and has various operations about Routes.
   */
  constructor(private urlRoute: ActivatedRoute,
    private routeService: RouteService) { }

  ngOnInit(): void {
    this.getRoute();
  }

  /**
   * Using the id passed by the URL route, gets the route in the database whose id corresponds.
   */
  getRoute(): void {
    const id = String(this.urlRoute.snapshot.paramMap.get("id"));
    this.routeService.getRoute(id)
      .subscribe((route) => {
        //Isto é feito porque o angular é bue fixe e não usa o construtor das classes, então o Date nas route.stops é na realidade apenas uma string...
        var stops: Array<LinkedStop> = [];
        route.stops?.forEach((s) => {
          stops.push(new LinkedStop(s.id, s.stopId, s.routeId, new Date(s.time), s.day, s.stop, s.route))
        })
        this.route = new Route(route.routeId, route.transportNumber ?? null, route.name, route.description,
          route.transportType, route.companyName, stops);
      });
  }
}
