import { Component, OnInit, ViewChild } from '@angular/core';
import { Route } from '../route';
import { RouteService } from '../route.service';
import { TransportType } from '../transport-type'
import { SelectionModel } from '@angular/cdk/collections';
import { MatTableDataSource } from '@angular/material/table';
import { MatPaginator } from '@angular/material/paginator';

@Component({
  selector: 'app-route-management',
  templateUrl: './route-management.component.html',
  styleUrls: ['./route-management.component.css']
})
/**
 * Class for the component of the page of Route Management
 */
export class RouteManagementComponent implements OnInit {
  /**
   * The paginator used in the route management table.
   */
  @ViewChild(MatPaginator) paginator!: MatPaginator;
  /**
   * The columns to be displayed in the route management table.
   */
  displayedColumns: string[] = ['select', 'number', 'name', 'type', 'company', 'duration', 'details'];
  /**
   * The data source used in the route management table.
   */
  dataSource!: MatTableDataSource<Route>
  /**
   * Holds the routes selected via checkbox.
   */
  selection = new SelectionModel<Route>(true, []);
  /**
   * The enum for reverse mapping in the html.
   */
  TransportType = TransportType;
  /**
   * All the existent routes.
   */
  public routes: Route[] = [];

  /**
   * Constructor for the component.
   * Gets the routes in the system and add them to the "routes" attribute.
   * @param {RouteService} routeService Service that communicates with the server and has various operations about Routes.
   */
  constructor(private routeService: RouteService) {
    this.routeService = routeService;
  }

  ngOnInit(): void {
    this.routeService.getRoutes().subscribe((rts) => {
      rts.forEach((route) => {
        this.routes.push(new Route(route.routeId, route.transportNumber, route.name, route.description,
          route.transportType, route.companyName, route.stops));
      });
      this.dataSource = new MatTableDataSource<Route>(this.routes);
      this.dataSource.paginator = this.paginator;
    });
  }

  /**
   * Deletes all the checked routes.
   */
  deleteRoutes() {
    this.selection.selected.forEach((route) => {
      var routeToDeleteId = route.routeId;
      this.routeService.deleteRoute(routeToDeleteId).subscribe(result => {
        const index = this.routes.findIndex((route) => route.routeId == routeToDeleteId);
        this.routes.splice(index, 1);
        this.dataSource.data = this.routes;
      });
    });
  }

  /**
   *  Checks if the number of selected elements matches the total number of rows.
   */
  isAllSelected() {
    const numSelected = this.selection.selected.length;
    const numRows = this.dataSource.data.length;
    return numSelected === numRows;
  }

  /**
   *  Selects all rows if they are not all selected; otherwise clear selection.
   */
  toggleAllRows() {
    if (this.isAllSelected()) {
      this.selection.clear();
      return;
    }
    this.selection.select(...this.dataSource.data);
  }

  /**
   * Aplies a filter to the list of existing routes.
   * @param {Event} event The parameter to filter the list for.
   */
  applyFilter(event: Event) {
    const filterValue = (event.target as HTMLInputElement).value;
    this.dataSource.filter = filterValue.trim().toLowerCase();
  }
}
